		LJMP START
		ORG 100H

START:
	LCALL LCD_CLR
	
	
    LCALL WAIT_KEY
	MOV R0,A
	LCALL BCD
	LCALL WRITE_HEX	
	
	MOV A,#'-'
	LCALL WRITE_DATA
	
	LCALL WAIT_KEY
	MOV R1,A
	LCALL NEG
	LCALL WRITE_HEX
	
	MOV A,#'='
	LCALL WRITE_DATA
	
	MOV A,R0
	ADD A,R1
	LCALL WRITE_HEX
	
	


	SJMP $
BCD:
	MOV B,#10
	
	DIV AB
	SWAP A
	ADD A,B
	
	RET
	
NEG:
	CPL A
	INC A
	
	RET
	
	RET
	NOP
