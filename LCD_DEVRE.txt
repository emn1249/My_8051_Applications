	MOV A,R7
	LCALL WRITE_HEX
	
		MOV A,#10
	LCALL DELAY_100MS

	CLR RS0
	MOV R7,#3
	MOV A,R7
	LCALL WRITE_HEX
	
		MOV A,#10
	LCALL DELAY_100MS
	LJMP START
	RET	
	NOP
